#include <iostream>

using namespace std;

int main()
{
    //OK if between 68 and 72
    bool tempOK;

    cout << "Is the temperature OK? (1 = yes, 0 = no)" << endl;

    cin >> tempOK;

    if(true == tempOK)
        cout << "Don't touch that dial!" << endl;

    if(false == tempOK)
        cout << "Adjust the thermostat!" << endl;


    return 0;

}
