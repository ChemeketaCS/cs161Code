#include <iostream>
#include <cmath>

using namespace std;

int main()
{
    cout << "Enter temperature: ";

    int temperature;
    cin >> temperature;

    //temperature should be between 68 and 72

    if( temperature <= 72 ) {
        if( temperature >= 68 ) {
            cout << "Temp is OK!";
        }
        else
            cout << "Turn it up!";
    }
    else
        cout << "Turn it down!";

    return 0;

}
