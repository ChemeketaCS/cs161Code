#include <iostream>

using namespace std;


int main()
{

    double score = 76;


    if (score >= 90.0)
      cout << "Grade is A" << endl;
    else if (score >= 80.0)
      cout << "Grade is B" << endl;
    else if (score >= 70.0)
      cout << "Grade is C" << endl;
    else if (score >= 60.0)
      cout << "Grade is D" << endl;
    else
      cout << "Grade is F" << endl;


}
