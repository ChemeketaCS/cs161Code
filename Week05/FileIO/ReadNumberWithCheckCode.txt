#include <iostream>
#include <fstream>

using namespace std;

int main()
{
    ifstream inFile;
    inFile.open("Numbers.txt");

    if(inFile.fail()) {
        cout << "Could not open file" << endl;
        return 0;
    }

    int number;
    inFile >> number;   //read number from file

    if(inFile.fail()) {
        cout << "Could not read a number" << endl;
        return 0;
    }

    //guess it worked
    cout << number << endl;

    return 0;
}
