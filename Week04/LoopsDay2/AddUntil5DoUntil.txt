#include <iostream>

using namespace std;

int main()
{

    int sum = 0;

    do
    {
        cout << "Enter an integer (the input ends "
             << "if it is 0): ";

        int data;
        cin >> data;

        //add this number... depends on 0 not affecting result!
        sum += data;

    }
    while (data != 0);

    cout << "The sum is " << sum << endl;

}
