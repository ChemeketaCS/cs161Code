#include <iostream>
#include <cmath>
using namespace std;

int main()
{

    cout << "Find all prime numbers <= n, enter n: ";
    int n;
    cin >> n;

    cout << "The prime numbers are:" << endl;

    int number = 2; // number we are testing for primeness
    while (number <= n)
    {
        cout << number << " ";

        number++;
    }

    cout << endl;

    return 0;
}
