#include <iostream>

using namespace std;

int main() {
  const int MYSTERY_VALUE = 120;

  int number = MYSTERY_VALUE;
  int curNumber = number - 1;

  while (curNumber >= 1) {
    if (number % curNumber == 0) break;

    // never get here if we break...
    curNumber--;
  }

  cout << "Largest factor is " << curNumber << endl;
}
