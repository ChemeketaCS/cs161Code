#include <iostream>

using namespace std;

int main()
{
    int i = 1;
    int total = 0;

    while(i <= 100) {
        total += i;     //add current value
        i++;            //go to next value
    }

    int count = i - 1; //counter went one too far
    double average = static_cast<double>(total) / count;

    cout << average << endl;

    return 0;
}
